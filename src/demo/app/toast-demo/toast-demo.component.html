VxToast

Inject the VxToast service into your app to create toasts. 
{{toastUsage}}

Methods

open(options: CreateToastOptions): VxToastComponent;

CreateToastOptions

/** Required, the text to put in the toast */
text: string;
/** Where to place the toast, the default is "top-right" */
position: ToastPosition;
/** How long to show the toast in milliseconds, defaults to 3000 */
duration: number;
/** What style of toast to show, defaults to "success" */
type: ToastType;
/** An optional title for the toast. */
title: string;
/** Whether or not to show the close toast button.  Defaults to true */
showClose: boolean;

ToastType

export type ToastType = 'error' | 'info' | 'success' | 'warn';

ToastPosition

export type ToastPosition = 'top-left' | 'top-middle' | 'top-right' | 'bottom-left' | 'bottom-middle' | 'bottom-right';
